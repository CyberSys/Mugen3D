local FSMS = {
    "Lua/FSM/CharFSMCommon",
    "Chars/Origin/fsm/OriginFSM_N",
    "Chars/Origin/fsm/OriginFSM_H",
}

local M = {}

function M.__loadall()
    for _, name in pairs(FSMS) do
        local t = require(name)
        for k, v in pairs(t) do
            M[k] = v
        end
    end
end

return M
