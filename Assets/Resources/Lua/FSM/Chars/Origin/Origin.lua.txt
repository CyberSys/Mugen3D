local Char = require "FSM/Chars/Char"
local fsm = require "FSM/Chars/Origin/OriginFSM"
local config = require "FSM/Chars/Origin/OriginCfg"
fsm.__loadall()


local M = {} 

function M:init(player)
	local obj = Char.new(player)
	obj.fsm = fsm
	obj.cfg = cfg
	setmetatable(self, {__index = obj})
end

return M
