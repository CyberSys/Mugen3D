local M = {}

local function registerEvent(self)
    self.btnSingleVS.onClick:AddListener(function()
    	print("click SingleVS")
	    UIManager.PushView(UIDef.ViewCharacterSelect, self.transform.parent)
	end)
	self.btnTrain.onClick:AddListener(function()
		print("click train")
	end)
end

function M:Init(view)
	self.view = view
	self.transform = view.transform
	self.btnSingleVS = self.transform:Find("VerticalGroup/SingleVS"):GetComponent("Button")
	self.btnTrain = self.transform:Find("VerticalGroup/Train"):GetComponent("Button")
	self.btnOption = self.transform:Find("VerticalGroup/Options"):GetComponent("Button")
	self.btnExit = self.transform:Find("VerticalGroup/Exit"):GetComponent("Button")
	registerEvent(self)	
end

function M:Update()
    --print("lua update")
end

return M
